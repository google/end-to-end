52/**
53 * Implements a "context". Provides a high level API for key management,
54 * encryption and signing. This context is used by external code, such as the
55 * extension's user interface, to call the base OpenPGP library.
56 * @constructor
57 * @param {!e2e.openpgp.KeyManager} keyManager The Key Manager object.
58 * @implements {e2e.openpgp.Context2}
59 */
60e2e.openpgp.Context2Impl = function(keyManager) {
61 /**
62 * Key Manager object.
63 * @private {!e2e.openpgp.KeyManager}
64 */
65 this.keyManager_ = keyManager;
66 /**
67 * Should the output of the functions be ASCII-armored.
68 * @private {boolean}
69 */
70 this.armorOutput_ = true;
71 /**
72 * List of headers to add to armored messages (Version, Comment, etc).
73 * @type {!Object.<string>}
74 * @private
75 */
76 this.armorHeaders_ = {};
77};

267/**
268 * Selects the correct encrypt/sign operation and calls it, returning the
269 * results.
270 * @param {!e2e.openpgp.Plaintext} plaintext The plaintext.
271 * @param {!e2e.openpgp.EncryptOptions} options Metadata to add.
272 * @param {!Array.<!e2e.openpgp.Key>} encryptionKeys The keys to
273 * encrypt the message with.
274 * @param {!Array.<string>} passphrases Passphrases to use for symmetric
275 * key encryption of the message.
276 * @param {!Array.<!e2e.openpgp.packet.SurrogateSecretKey>} signKeys
277 * @return {!e2e.openpgp.Context2Impl.EncodableMessagePromise_}
278 * The encrypted and/or signed message, before encoding.
279 * @private
280 */
281e2e.openpgp.Context2Impl.prototype.doEncryptSignOperation_ = function(
282 plaintext, options, encryptionKeys, passphrases, signKeys) {
283 if (signKeys.length > 1) {
284 // TODO(koto): Support multiple signing keys.
285 throw new e2e.openpgp.error.UnsupportedError(
286 'Signing with multiple keys is unsupported.');
287 }
288 var needsEncryption = (encryptionKeys.length > 0 || passphrases.length > 0);
289 var result;
290 if (needsEncryption) {
291 // Encryption required.
292 result = this.encryptSign_(
293 plaintext,
294 options,
295 encryptionKeys,
296 passphrases,
297 signKeys);
298 } else {
299 // No encryption required, just signing.
300 if (signKeys.length == 0) {
301 throw new e2e.openpgp.error.InvalidArgumentsError(
302 'No signing keys provided.');
303 }
304
305 if (typeof plaintext == 'string') {
306 // Clearsign messages support only single signature keys now.
307 result = this.clearSign_(plaintext, signKeys[0]);
308 } else {
309 result = this.byteSign_(plaintext, signKeys);
310 }
311 }
312
313 // Creating a type annotation here to satisfy the covariance/contravariance
314 // problems of Thenables. Do NOT use this result object outside this class.
315 return /** @type {!e2e.openpgp.Context2Impl.EncodableMessagePromise_} */ (
316 result);
317};

320/**
321 * Internal implementation of the encrypt/sign operation.
322 * @param {!e2e.ByteArray|string} plaintext The plaintext.
323 * @param {!e2e.openpgp.EncryptOptions} options Metadata to add.
324 * @param {!e2e.openpgp.Keys} encryptionKeys The key handles to encrypt the
325 * message to.
326 * @param {!Array.<string>} passphrases Passphrases to use for symmetric
327 * key encryption of the message.
328 * @param {!Array.<!e2e.openpgp.packet.SurrogateSecretKey>} signatureKeys
329 * The keys to sign the message with.
330 * @private
331 * @return {!goog.Thenable<!e2e.openpgp.block.EncryptedMessage>} The encrypted
332 * message.
333 */
334e2e.openpgp.Context2Impl.prototype.encryptSign_ = function(
335 plaintext, options, encryptionKeys, passphrases, signatureKeys) {
336 var keys = goog.array.map(encryptionKeys, this.requirePublicKey_, this);
337 var literal = e2e.openpgp.block.LiteralMessage.construct(plaintext);
338 goog.asserts.assert(signatureKeys.length <= 1);
339 return e2e.openpgp.block.EncryptedMessage.construct(
340 literal,
341 keys,
342 passphrases,
343 signatureKeys[0] || undefined);
344};

347/**
348 * Converts a public key handle to a key block.
349 * Throws an error when a key handle does not represent a public key or the
350 * resulting key is invalid due to e.g. invalid/outdated signatures.
351 * @param {!e2e.openpgp.Key} key A key handle.
352 * @return {!e2e.openpgp.block.TransferablePublicKey}
353 * @private
354 */
355e2e.openpgp.Context2Impl.prototype.requirePublicKey_ = function(key) {
356 goog.asserts.assert(!key.key.secret);
357 var keyBlock = e2e.openpgp.block.factory.parseByteArrayTransferableKey(
358 key.serialized);
359 if (keyBlock instanceof e2e.openpgp.block.TransferablePublicKey) {
360 keyBlock.processSignatures();
361 return keyBlock;
362 }
363 throw new e2e.openpgp.error.InvalidArgumentsError('Invalid public key.');
364};

411/**
412 * Internal implementation of the byte sign operation.
413 * @param {!e2e.ByteArray|string} plaintext The plaintext.
414 * @param {!Array<!e2e.openpgp.packet.SurrogateSecretKey>} sigKeys The keys to
415 * sign the message with.
416 * @private
417 * @return {!goog.Thenable<!e2e.openpgp.block.Message>}
418 */
419e2e.openpgp.Context2Impl.prototype.byteSign_ = function(
420 plaintext, sigKeys) {
421 var msg = e2e.openpgp.block.LiteralMessage.construct(plaintext);
422 goog.asserts.assert(sigKeys.length <= 1);
423 var sigKey = sigKeys[0];
424 if (!sigKey) {
425 throw new e2e.openpgp.error.InvalidArgumentsError('Invalid signing key.');
426 }
427 return msg.signWithOnePass(sigKey).then(function() {
428 return msg;
429 });
430};

433/**
434 * Encodes the encrypted and/or signed message for output.
435 * @param {!e2e.openpgp.block.Message|!e2e.openpgp.ClearSignMessage} message
436 * The message to encode.
437 * @return {!e2e.ByteArray|string} The message, optionally armored.
438 * @private
439 */
440e2e.openpgp.Context2Impl.prototype.encodeMessage_ = function(message) {
441 // Special encoding for clearsigned messages (they are not block.Message)
442 if (message instanceof e2e.openpgp.ClearSignMessage) {
443 if (this.armorOutput_) {
444 return e2e.openpgp.asciiArmor.encodeClearSign(
445 message, this.armorHeaders_);
446 }
447 // Convert to literal message to output a byte serialization.
448 message = message.toLiteralMessage();
449 }
450 var bytes = message.serialize();
451 if (this.armorOutput_) {
452 return e2e.openpgp.asciiArmor.encode(
453 'MESSAGE', goog.asserts.assertArray(bytes),
454 this.armorHeaders_);
455 }
456 return bytes;
457};

460/** @override */
461e2e.openpgp.Context2Impl.prototype.verifyDecrypt = function(encryptedMessage,
462 passphraseCallback, opt_decryptionKeys, opt_verificationKeys) {
463 return goog.Promise.resolve(undefined).then(function() {
464 var encryptedData, charset;
465 if (typeof encryptedMessage == 'string') {
466 if (e2e.openpgp.asciiArmor.isClearSign(encryptedMessage)) {
467 return this.verifyClearSign_(
468 e2e.openpgp.asciiArmor.parseClearSign(encryptedMessage),
469 opt_verificationKeys);
470 }
471 var armoredMessage = e2e.openpgp.asciiArmor.parse(encryptedMessage);
472 charset = armoredMessage.charset;
473 encryptedData = armoredMessage.data;
474 } else {
475 encryptedData = encryptedMessage;
476 }
477 return this.verifyDecryptInternal(
478 encryptedData, passphraseCallback, charset, opt_decryptionKeys,
479 opt_verificationKeys);
480 }, null, this);
481};

505/**
506 * Internal implementation of the verification/decryption operation.
507 * @param {!e2e.ByteArray} encryptedMessage The encrypted data.
508 * @param {function(string):!goog.Thenable<string>} passphraseCallback This
509 * callback is used for requesting an action-specific passphrase from the
510 * user.
511 * @param {string=} opt_charset The (optional) charset to decrypt with.
512 * @param {!Array.<!e2e.openpgp.Key>=} opt_decryptionKeys If present,
513 * only those keys will be used for decryption. Otherwise, Context2 uses
514 * Key ID hints in the message to resolve keys on its own.
515 * @param {!Array.<!e2e.openpgp.Key>=} opt_verificationKeys If present,
516 * only those keys will be used for signature verification. Otherwise,
517 * Key ID hints in the message will be used to resolve the keys.
518 * @protected
519 * @return {!e2e.openpgp.VerifyDecryptPromise}
520 */
521e2e.openpgp.Context2Impl.prototype.verifyDecryptInternal = function(
522 encryptedMessage, passphraseCallback, opt_charset, opt_decryptionKeys,
523 opt_verificationKeys) {
524 var block = e2e.openpgp.block.factory.parseByteArrayMessage(
525 encryptedMessage, opt_charset);
526 if (block instanceof e2e.openpgp.block.EncryptedMessage) {
527 var keyCipherCallback = goog.bind(function(keyId, algorithm) {
528 return goog.async.Deferred.fromPromise(
529 this.getDecryptionCipher_(keyId, algorithm, opt_decryptionKeys));
530 }, this);
531
532 var pwCallbackWrapper = function(passphrase) {
533 return goog.async.Deferred.succeed(passphrase).addCallback(
534 passphraseCallback);
535 };
536
537 return block.decrypt(keyCipherCallback, pwCallbackWrapper)
538 .addCallbacks(function(message) {
539 return this.processLiteralMessage_(message, opt_verificationKeys);
540 }, goog.Promise.reject, this)
541 .then(function(result) {
542 result.decrypt.wasEncrypted = true;
543 return result;
544 });
545 } else {
546 return this.processLiteralMessage_(block, opt_verificationKeys)
547 .then(function(result) {
548 result.decrypt.wasEncrypted = false;
549 return result;
550 });
551 }
552};

555/**
556 * Returns a decrypting cipher for a first found secret key with a given Key ID
557 * Optionally limits the search to a set of key object in a provided array.
558 * @param {!e2e.openpgp.KeyId} keyId Key ID hint.
559 * @param {!e2e.cipher.Algorithm} algorithm Algorithm hint.
560 * @param {!Array.<!e2e.openpgp.Key>=} opt_allowedDecryptionKeys If provided,
561 * only ciphers for keys in that array can be returned.
562 * @return {!goog.Promise<e2e.openpgp.KeyProviderCipher>} The cipher, or null
563 * if no matching cipher has been found.
564 * @private
565 */
566e2e.openpgp.Context2Impl.prototype.getDecryptionCipher_ = function(keyId,
567 algorithm, opt_allowedDecryptionKeys) {
568 // Currently only the first key matching a Key ID is returned
569 // Key ID collisions in secret keys can prevent decryption.
570 return goog.Promise.resolve().then(function() {
571 if (goog.isArray(opt_allowedDecryptionKeys)) {
572 // Preselect based on the key object
573 var foundKey = goog.array.find(opt_allowedDecryptionKeys, function(key) {
574 return (key.key.secret &&
575 key.decryptionAlgorithm == algorithm &&
576 e2e.compareByteArray(goog.asserts.assertArray(key.decryptionKeyId),
577 keyId));
578 });
579 return foundKey ? [foundKey] : [];
580 } else {
581 // Use the Key ID hint to resolve the key.
582 return this.keyManager_.getKeysByKeyId(
583 e2e.openpgp.KeyPurposeType.DECRYPTION, keyId);
584 }
585 }, null, this).then(function(keyObjects) {
586 if (keyObjects.length == 0) {
587 return null;
588 }
589 if (keyObjects.length > 1) {
590 throw new e2e.openpgp.error.UnsupportedError(
591 'Multiple decryption keys are not supported.');
592 }
593 return new e2e.openpgp.KeyProviderCipher(
594 algorithm,
595 undefined,
596 goog.bind(
597 this.keyManager_.decrypt,
598 this.keyManager_,
599 keyObjects[0],
600 keyId));
601 }, null, this);
602};

605/**
606 * Processes a literal message and returns the result of verification.
607 * @param {e2e.openpgp.block.Message} block
608 * @param {!Array.<!e2e.openpgp.Key>=} opt_verificationKeys If present,
609 * only those keys will be used for signature verification. Otherwise,
610 * Key ID hints in the message will be used to resolve the keys.
611 * @return {!e2e.openpgp.VerifyDecryptPromise}
612 * @private
613 */
614e2e.openpgp.Context2Impl.prototype.processLiteralMessage_ = function(block,
615 opt_verificationKeys) {
616 var literalBlock = block.getLiteralMessage();
617 var verifyPromise = goog.Promise.resolve(null);
618 if (literalBlock.signatures) {
619 verifyPromise = this.verifyMessage_(literalBlock, opt_verificationKeys);
620 }
621 return verifyPromise.then(function(verify) {
622 return /** @type {!e2e.openpgp.VerifiedDecrypt} */ ({
623 'decrypt': {
624 'data': new Uint8Array(literalBlock.getData()),
625 'options': {
626 'charset': literalBlock.getCharset(),
627 'creationTime': literalBlock.getTimestamp(),
628 'filename': literalBlock.getFilename()
629 },
630 'wasEncrypted': false
631 },
632 'verify': verify
633 });
634 });
635};

638/**
639 * Verifies signatures places on a LiteralMessage
640 * @param {!e2e.openpgp.block.LiteralMessage} message Block to verify
641 * @param {!Array.<!e2e.openpgp.Key>=} opt_verificationKeys If present,
642 * only those keys will be used for signature verification. Otherwise,
643 * Key ID hints in the message will be used to resolve the keys.
644 * @return {!goog.Thenable<e2e.openpgp.VerifyResult>} Verification result.
645 * @private
646 */
647e2e.openpgp.Context2Impl.prototype.verifyMessage_ = function(
648 message, opt_verificationKeys) {
649 var keyBlocksPromise;
650
651 if (goog.isDefAndNotNull(opt_verificationKeys)) {
652 // We know the keys upfront.
653 keyBlocksPromise = goog.Promise.resolve(
654 goog.array.map(opt_verificationKeys, this.requirePublicKey_, this));
655 } else {
656 // Get keys matching key IDs declared in signatures.
657 keyBlocksPromise = goog.Promise.all(
658 goog.array.map(message.getSignatureKeyIds(), goog.bind(function(keyId) {
659 return this.keyManager_.getKeysByKeyId(
660 e2e.openpgp.KeyPurposeType.VERIFICATION, keyId);
661 }, this))).then(function(keyHandles) {
662 var foundKeys = goog.array.flatten(goog.array.filter(
663 keyHandles, function(keyHandle) {
664 return !goog.isNull(keyHandle);
665 }));
