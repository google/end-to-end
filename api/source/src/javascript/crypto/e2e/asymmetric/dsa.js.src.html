114/**
115 * Sets the DSA public key and/or private key.
116 * @override
117 */
118e2e.signer.Dsa.prototype.setKey = function(keyArg, opt_keySize) {
119 var key = /** @type {!e2e.signer.key.Dsa} */ (keyArg);
120 goog.asserts.assertArray(key['p'], 'The prime modulus should be defined.');
121 this.p_ = new e2e.BigPrimeNum(key['p']);
122 var lenP = this.p_.getBitLength();
123 goog.asserts.assertArray(key['q'], 'The prime order should be defined.');
124 this.q_ = new e2e.BigPrimeNum(key['q']);
125 var lenQ = this.q_.getBitLength();
126
127 switch (lenP) {
128 case 1024:
129 if (lenQ != 160) {
130 throw new e2e.openpgp.error.InvalidArgumentsError(
131 'q must be 160-bit when p is 1024-bit.');
132 }
133 this.hash_ = e2e.hash.factory.require(
134 e2e.hash.Algorithm.SHA1);
135 break;
136 case 2048:
137 if (lenQ == 224) {
138 this.hash_ = e2e.hash.factory.require(
139 e2e.hash.Algorithm.SHA224);
140 } else if (lenQ == 256) {
141 this.hash_ = e2e.hash.factory.require(
142 e2e.hash.Algorithm.SHA256);
143 } else {
144 throw new e2e.openpgp.error.InvalidArgumentsError(
145 'q must be 224-bit or 256-bit when p is 2048-bit.');
146 }
147 break;
148 case 3072:
149 if (lenQ != 256) {
150 throw new e2e.openpgp.error.InvalidArgumentsError(
151 'q must be 256-bit when p is 3072-bit.');
152 }
153 this.hash_ = e2e.hash.factory.require(
154 e2e.hash.Algorithm.SHA256);
155 break;
156 default:
157 throw new e2e.openpgp.error.UnsupportedError(
158 'The bit lengths of p and q are not supported.');
159 break;
160 }
161
162 var pminus1 = this.p_.subtract(e2e.BigNum.ONE);
163 // q should be a divisor of p - 1.
164 if (this.q_.mod(pminus1).isEqual(
165 e2e.BigNum.ZERO)) {
166 throw new e2e.openpgp.error.InvalidArgumentsError(
167 'q must be a divisor of p - 1.');
168 }
169
170 goog.asserts.assertArray(key['g'], 'The generator should be defined.');
171 this.g_ = new e2e.BigNum(key['g']);
172 // 1 < g < p and g^q (mod p) == 1.
173 if (!this.g_.isBetween(e2e.BigNum.ONE, this.p_) ||
174 !this.p_.modPower(this.g_, key['q']).isEqual(e2e.BigNum.ONE)) {
175 throw new e2e.openpgp.error.InvalidArgumentsError(
176 'Invalid generator.');
177 }
178
179 if (!goog.isDefAndNotNull(key['x']) &&
180 !goog.isDefAndNotNull(key['y'])) {
181 goog.asserts.fail('Either public key or private key should be defined.');
182 }
183
184 if (goog.isDefAndNotNull(key['x'])) {
185 this.x_ = new e2e.BigNum(key['x']);
186 if (!this.x_.isBetween(e2e.BigNum.ZERO, this.q_)) {
187 throw new e2e.openpgp.error.InvalidArgumentsError(
188 'x must be in range (0, q).');
189 }
190 }
191
192 if (goog.isDefAndNotNull(key['y'])) {
193 this.y_ = new e2e.BigNum(key['y']);
194 // NIST SP 800-89 checks for DSA.
195 // 1 < y < p-1
196 if (!this.y_.isBetween(e2e.BigNum.ONE, pminus1)) {
197 throw new e2e.openpgp.error.InvalidArgumentsError(
198 'y must be in the range(1, p-1).');
199 }
200 // y^q = 1 (mod p).
201 if (!this.p_.modPower(this.y_, this.q_).isEqual(e2e.BigNum.ONE)) {
202 throw new e2e.openpgp.error.InvalidArgumentsError(
203 'Invalid public key.');
204 }
205 if (goog.isDefAndNotNull(key['x'])) {
206 // y == g^x (mod p).
207 if (!this.p_.modPower(this.g_, key['x']).isEqual(this.y_)) {
208 throw new e2e.openpgp.error.InvalidArgumentsError(
209 'Invalid public key.');
210 }
211 }
212 }
213
214 // Save key material to serialize later the key.
215 goog.base(this, 'setKey', key);
216};

219/** @inheritDoc */
220e2e.signer.Dsa.prototype.sign = function(m) {
221 /** @type {!e2e.signer.signature.Signature} */
222 var sig;
223 do {
224 var k = this.generatePerMessageSecret_();
225 sig = this.signWithNonce_(m, k);
226 var r = new e2e.BigNum(sig['r']);
227 var s = new e2e.BigNum(sig['s']);
228 } while (r.isEqual(e2e.BigNum.ZERO) ||
229 s.isEqual(e2e.BigNum.ZERO));
230 return e2e.async.Result.toResult(sig);
231};

246/** @inheritDoc */
247e2e.signer.Dsa.prototype.verify = function(m, sig) {
248 goog.asserts.assertObject(this.p_, 'The prime modulus should be defined.');
249 goog.asserts.assertObject(this.q_, 'The prime order should be defined.');
250 goog.asserts.assertObject(this.g_, 'The order should be defined.');
251 goog.asserts.assertObject(this.y_, 'The public key should be defined.');
252
253 var r = new e2e.BigNum(sig['r']);
254 var s = new e2e.BigNum(sig['s']);
255 if (!r.isBetween(e2e.BigNum.ZERO, this.q_) ||
256 !s.isBetween(e2e.BigNum.ZERO, this.q_)) {
257 return e2e.async.Result.toResult(false);
258 }
259
260 var w = this.q_.modInverse(s);
261 var z = new e2e.BigNum(this.hash_.hash(m));
262 var u1 = this.q_.modMultiply(z.mod(this.q_), w); // z may be >= q_
263 var u2 = this.q_.modMultiply(r, w);
264 var v = this.p_.modMultiply(this.p_.modPower(this.g_, u1),
265 this.p_.modPower(this.y_, u2)).mod(this.q_);
266 return e2e.async.Result.toResult(v.isEqual(r));
267};

270/**
271 * Generates the DSA signature using the provided per-message secret.
272 * @param {!e2e.ByteArray} m The message to be signed.
273 * @param {!e2e.BigNum} k The per-message secret.
274 * @return {!e2e.signer.signature.Signature}
275 * @private
276 */
277e2e.signer.Dsa.prototype.signWithNonce_ = function(m, k) {
278 goog.asserts.assertObject(this.p_, 'The prime modulus should be defined.');
279 goog.asserts.assertObject(this.q_, 'The prime order should be defined.');
280 goog.asserts.assertObject(this.g_, 'The order should be defined.');
281 goog.asserts.assertObject(this.x_, 'The private key should be defined.');
282
283 // Sanity check on the per-message nonce that it's in [1, q-1].
284 if (!k.isBetween(e2e.BigNum.ZERO, this.q_)) {
285 throw new e2e.openpgp.error.InvalidArgumentsError(
286 'Failed to sign message: invalid per-message nonce.');
287 }
288 // r = (g^k mod p) mod q.
289 var r = this.p_.modPower(this.g_, k).mod(this.q_);
290 var hashValue = this.hash_.hash(m);
291 var z = new e2e.BigNum(hashValue);
292 // s = (k^{-1} (z + xr)) mod q.
293 var tmp = z.add(this.q_.modMultiply(this.x_, r)).mod(this.q_);
294 var s = this.q_.modMultiply(this.q_.modInverse(k), tmp);
295 return {
296 'r': r.toByteArray(),
297 's': s.toByteArray(),
298 'hashValue': hashValue
299 };
300};

303/**
304 * Generates a random number used as the per-message secret in DSA.
305 * @return {!e2e.BigNum}
306 * @private
307 */
308e2e.signer.Dsa.prototype.generatePerMessageSecret_ = function() {
309 goog.asserts.assertObject(this.q_,
310 'Cannot generate random per-message secret: q should be defined.');
311
312 // 64 more bits are requested from the PRNG than are needed for this nonce
313 // to avoid bias in the modular reduction in the last step of this function.
314 // Otherwise this might leak a fraction of a bit of the nonce, and that's
315 // enough for Bleichenbacher to steal the private key.
316 var nonceLength = Math.ceil((this.q_.getBitLength() + 64) / 8);
317 // OpenPGP supports only 3 key sizes in bits: 160, 224 and 256, so double
318 // check nonce length to ensure enough entropy shall be requested from the
319 // PRNG.
320 if (nonceLength != 28 /* 160-bit q */ &&
321 nonceLength != 36 /* 224-bit q */ &&
322 nonceLength != 40 /* 256-bit q */) {
323 throw new e2e.openpgp.error.InvalidArgumentsError(
324 'Cannot generate random nonce: invalid nonce length.');
325 }
326 var nonce = new e2e.BigNum(
327 e2e.random.getRandomBytes(nonceLength));
328 // nonce is [1, q - 1].
329 return nonce.mod(this.q_.subtract(e2e.BigNum.ONE)).add(
330 e2e.BigNum.ONE);
331};
