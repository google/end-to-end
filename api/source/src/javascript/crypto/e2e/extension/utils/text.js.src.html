37/**
38 * Formats a body of text such that all lines do not exceed a given length.
39 * @param {string} str The body of text to wrap around.
40 * @param {number} maxlen The maximum length of a line.
41 * @return {string} The formatted text.
42 */
43utils.prettyTextWrap = function(str, maxlen) {
44 var regexp = new RegExp(goog.string.format('(.{%d})', maxlen), 'g');
45 str = str.trim().replace(regexp, '$1\n');
46
47 var carry = '';
48 var lines = goog.array.map(str.split('\n'), function(line) {
49 var newline = goog.string.format('%s%s', carry, line).trim();
50 carry = '';
51
52 if (newline.length >= maxlen && !goog.string.endsWith(newline, ' ')) {
53 var lastSpace = newline.lastIndexOf(' ');
54 if (lastSpace > -1 &&
55 goog.string.isAlphaNumeric(newline.substring(newline.length - 1))) {
56 carry = newline.substring(lastSpace + 1);
57 return newline.substring(0, lastSpace);
58 }
59 }
60
61 return newline;
62 });
63
64 if (carry) {
65 goog.array.extend(lines, utils.prettyTextWrap(carry, maxlen).split('\n'));
66 }
67
68 return lines.join('\n');
69};

72/**
73 * Determines the requested PGP action based on the content that the user has
74 * selected.
75 * @param {string} content The content that the user has selected.
76 * @return {constants.Actions} The requested PGP action.
77 */
78utils.getPgpAction = function(content) {
79 if (/^-----BEGIN PGP (?:SIGNED )?MESSAGE-----/.test(content)) {
80 return constants.Actions.DECRYPT_VERIFY;
81 }
82
83 if (/^-----BEGIN PGP PUBLIC KEY BLOCK-----/.test(content)) {
84 return constants.Actions.IMPORT_KEY;
85 }
86
87 if (/^-----BEGIN PGP PRIVATE KEY BLOCK-----/.test(content)) {
88 return constants.Actions.IMPORT_KEY;
89 }
90
91 return constants.Actions.ENCRYPT_SIGN;
92};

95/**
96 * Extract a valid e-mail address from 'user id <email>' string. If no valid
97 * e-mail address can be extracted, returns null. Uses
98 * goog.format.EmailAddress, but also enforces stricter rules.
99 * @param {string} recipient "username <email> string".
100 * @return {?string} Valid email address or null
101 */
102utils.extractValidEmail = function(recipient) {
103 var emailAddress = goog.format.EmailAddress.parse(recipient);
104 if (!emailAddress.isValid()) {
105 return null;
106 }
107 var email = emailAddress.getAddress();
108 if (!constants.EMAIL_ADDRESS_REGEXP.exec(emailAddress.getAddress())) {
109 return null;
110 }
111 return email;
112};

115/**
116 * Checks whether a URI is an HTTPS ymail origin.
117 * @param {!string} uri
118 * @return {boolean}
119 */
120utils.isYmailOrigin = function(uri) {
121 var googUri = new goog.Uri(uri);
122 return (googUri.getScheme() === 'https' &&
123 goog.string.endsWith(googUri.getDomain(), '.mail.yahoo.com'));
124};
