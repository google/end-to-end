118/**
119 * Returns a given value's contentDir property, constrained to a
120 * goog.i18n.bidi.Dir value or null. Returns null if the value is null,
121 * undefined, a primitive or does not have a contentDir property, or the
122 * property's value is not 1 (for LTR), -1 (for RTL), or 0 (for neutral).
123 *
124 * @param {*} value The value whose contentDir property, if any, is to
125 * be returned.
126 * @return {?goog.i18n.bidi.Dir} The contentDir property.
127 */
128soydata.getContentDir = function(value) {
129 if (value != null) {
130 switch (value.contentDir) {
131 case goog.i18n.bidi.Dir.LTR:
132 return goog.i18n.bidi.Dir.LTR;
133 case goog.i18n.bidi.Dir.RTL:
134 return goog.i18n.bidi.Dir.RTL;
135 case goog.i18n.bidi.Dir.NEUTRAL:
136 return goog.i18n.bidi.Dir.NEUTRAL;
137 }
138 }
139 return null;
140};

144 * Content of type {@link soydata.SanitizedContentKind.HTML}.
145 *
146 * The content is a string of HTML that can safely be embedded in a PCDATA
147 * context in your app. If you would be surprised to find that an HTML
148 * sanitizer produced {@code s} (e.g. it runs code or fetches bad URLs) and
149 * you wouldn't write a template that produces {@code s} on security or privacy
150 * grounds, then don't pass {@code s} here. The default content direction is
151 * unknown, i.e. to be estimated when necessary.
152 *
153 * @constructor
154 * @extends {goog.soy.data.SanitizedContent}
155 */
156soydata.SanitizedHtml = function() {
157 goog.soy.data.SanitizedContent.call(this); // Throws an exception.

164/**
165 * Returns a SanitizedHtml object for a particular value. The content direction
166 * is preserved.
167 *
168 * This HTML-escapes the value unless it is already SanitizedHtml or SafeHtml.
169 *
170 * @param {*} value The value to convert. If it is already a SanitizedHtml
171 * object, it is left alone.
172 * @return {!soydata.SanitizedHtml} A SanitizedHtml object derived from the
173 * stringified value. It is escaped unless the input is SanitizedHtml or
174 * SafeHtml.
175 */
176soydata.SanitizedHtml.from = function(value) {
177 // The check is soydata.isContentKind() inlined for performance.
178 if (value != null &&
179 value.contentKind === soydata.SanitizedContentKind.HTML) {
180 goog.asserts.assert(value.constructor === soydata.SanitizedHtml);
181 return /** @type {!soydata.SanitizedHtml} */ (value);
182 }
183 if (value instanceof goog.html.SafeHtml) {
184 return soydata.VERY_UNSAFE.ordainSanitizedHtml(
185 goog.html.SafeHtml.unwrap(value), value.getDirection());
186 }
187 return soydata.VERY_UNSAFE.ordainSanitizedHtml(
188 soy.esc.$$escapeHtmlHelper(String(value)), soydata.getContentDir(value));
189};

313/**
314 * Creates a factory for SanitizedContent types.
315 *
316 * This is a hack so that the soydata.VERY_UNSAFE.ordainSanitized* can
317 * instantiate Sanitized* classes, without making the Sanitized* constructors
318 * publicly usable. Requiring all construction to use the VERY_UNSAFE names
319 * helps callers and their reviewers easily tell that creating SanitizedContent
320 * is not always safe and calls for careful review.
321 *
322 * @param {function(new: T)} ctor A constructor.
323 * @return {!function(*, ?goog.i18n.bidi.Dir=): T} A factory that takes
324 * content and an optional content direction and returns a new instance. If
325 * the content direction is undefined, ctor.prototype.contentDir is used.
326 * @template T
327 * @private
328 */
329soydata.$$makeSanitizedContentFactory_ = function(ctor) {
330 /**
331 * @param {string} content
332 * @constructor
333 * @extends {goog.soy.data.SanitizedContent}
334 */
335 function InstantiableCtor(content) {
336 /** @override */
337 this.content = content;
338 }
339 InstantiableCtor.prototype = ctor.prototype;
340 /**
341 * Creates a ctor-type SanitizedContent instance.
342 *
343 * @param {*} content The content to put in the instance.
344 * @param {?goog.i18n.bidi.Dir=} opt_contentDir The content direction. If
345 * undefined, ctor.prototype.contentDir is used.
346 * @return {!goog.soy.data.SanitizedContent} The new instance. It is actually
347 * of type T above (ctor's type, a descendant of SanitizedContent), but
348 * there is no way to express that here.
349 */
350 function sanitizedContentFactory(content, opt_contentDir) {
351 var result = new InstantiableCtor(String(content));
352 if (opt_contentDir !== undefined) {
353 result.contentDir = opt_contentDir;
354 }
355 return result;
356 }
357 return sanitizedContentFactory;
358};

593/**
594 * Builds an augmented map. The returned map will contain mappings from both
595 * the base map and the additional map. If the same key appears in both, then
596 * the value from the additional map will be visible, while the value from the
597 * base map will be hidden. The base map will be used, but not modified.
598 *
599 * @param {!Object} baseMap The original map to augment.
600 * @param {!Object} additionalMap A map containing the additional mappings.
601 * @return {!Object} An augmented map containing both the original and
602 * additional mappings.
603 */
604soy.$$augmentMap = function(baseMap, additionalMap) {
605
606 // Create a new map whose '__proto__' field is set to baseMap.
607 /** @constructor */
608 function TempCtor() {}
609 TempCtor.prototype = baseMap;
610 var augmentedMap = new TempCtor();
611
612 // Add the additional mappings to the new map.
613 for (var key in additionalMap) {
614 augmentedMap[key] = additionalMap[key];
615 }
616
617 return augmentedMap;
618};

708 * compared and only the higher priority implementation is stored (if
709 * priorities are equal, an error is thrown).
710 *
711 * @param {string} delTemplateId The delegate template id.

718 delTemplateId, delTemplateVariant, delPriority, delFn) {

720 var mapKey = 'key_' + delTemplateId + ':' + delTemplateVariant;
721 var currPriority = soy.$$DELEGATE_REGISTRY_PRIORITIES_[mapKey];
722 if (currPriority === undefined || delPriority > currPriority) {
723 // Registering new or higher-priority function: replace registry entry.
724 soy.$$DELEGATE_REGISTRY_PRIORITIES_[mapKey] = delPriority;
725 soy.$$DELEGATE_REGISTRY_FUNCTIONS_[mapKey] = delFn;
726 } else if (delPriority == currPriority) {

728 throw Error(
729 'Encountered two active delegates with the same priority ("' +
730 delTemplateId + ':' + delTemplateVariant + '").');

732 // Registering lower-priority function: do nothing.

1078/**
1079 * Removes HTML tags from a string of known safe HTML.
1080 * If opt_tagWhitelist is not specified or is empty, then
1081 * the result can be used as an attribute value.
1082 *
1083 * @param {*} value The HTML to be escaped. May not be a string, but the
1084 * value will be coerced to a string.
1085 * @param {Object<string, boolean>=} opt_tagWhitelist Has an own property whose
1086 * name is a lower-case tag name and whose value is {@code 1} for
1087 * each element that is allowed in the output.
1088 * @return {string} A representation of value without disallowed tags,
1089 * HTML comments, or other non-text content.
1090 */
1091soy.$$stripHtmlTags = function(value, opt_tagWhitelist) {
1092 if (!opt_tagWhitelist) {
1093 // If we have no white-list, then use a fast track which elides all tags.
1094 return String(value).replace(soy.esc.$$HTML_TAG_REGEX_, '')
1095 // This is just paranoia since callers should normalize the result
1096 // anyway, but if they didn't, it would be necessary to ensure that
1097 // after the first replace non-tag uses of < do not recombine into
1098 // tags as in "<<foo>script>alert(1337)</<foo>script>".
1099 .replace(soy.esc.$$LT_REGEX_, '&lt;');
1100 }
1101
1102 // Escapes '[' so that we can use [123] below to mark places where tags
1103 // have been removed.
1104 var html = String(value).replace(/\[/g, '&#91;');
1105
1106 // Consider all uses of '<' and replace whitelisted tags with markers like
1107 // [1] which are indices into a list of approved tag names.
1108 // Replace all other uses of < and > with entities.
1109 var tags = [];
1110 var openListTags = [];
1111 var attrs = [];
1112 html = html.replace(
1113 soy.esc.$$HTML_TAG_REGEX_,
1114 function(tok, tagName) {
1115 if (tagName) {
1116 tagName = tagName.toLowerCase();
1117 if (opt_tagWhitelist.hasOwnProperty(tagName) &&
1118 opt_tagWhitelist[tagName]) {
1119 var isClose = tok.charAt(1) == '/';
1120 var index = tags.length;
1121 var start = '</';
1122 var attributes = '';
1123 if (!isClose) {
1124 start = '<';
1125 var match;
1126 while (match = soy.esc.$$HTML_ATTRIBUTE_REGEX_.exec(tok)) {
1127 if (match[1] && match[1].toLowerCase() == 'dir') {
1128 var dir = match[2];
1129 if (dir) {
1130 if (dir.charAt(0) == '\'' || dir.charAt(0) == '"') {
1131 dir = dir.substr(1, dir.length - 2);
1132 }
1133 dir = dir.toLowerCase();
1134 if (dir == 'ltr' || dir == 'rtl' || dir == 'auto') {
1135 attributes = ' dir="' + dir + '"';
1136 }
1137 }
1138 break;
1139 }
1140 }
1141 soy.esc.$$HTML_ATTRIBUTE_REGEX_.lastIndex = 0;
1142 }
1143 tags[index] = start + tagName + '>';
1144 attrs[index] = attributes;
1145 return '[' + index + ']';
1146 }
1147 }
1148 return '';
1149 });
1150
1151 // Escape HTML special characters. Now there are no '<' in html that could
1152 // start a tag.
1153 html = soy.esc.$$normalizeHtmlHelper(html);
1154
1155 var finalCloseTags = soy.$$balanceTags_(tags);
1156
1157 // Now html contains no tags or less-than characters that could become
1158 // part of a tag via a replacement operation and tags only contains
1159 // approved tags.
1160 // Reinsert the white-listed tags.
1161 html = html.replace(/\[(\d+)\]/g, function(_, index) {
1162 if (attrs[index] && tags[index]) {
1163 return tags[index].substr(0, tags[index].length - 1) + attrs[index] + '>';
1164 }
1165 return tags[index];
1166 });
1167
1168 // Close any still open tags.
1169 // This prevents unclosed formatting elements like <ol> and <table> from
1170 // breaking the layout of containing HTML.
1171 return html + finalCloseTags;
1172};

1189/**
1190 * Throw out any close tags that don't correspond to start tags.
1191 * If {@code <table>} is used for formatting, embedded HTML shouldn't be able
1192 * to use a mismatched {@code </table>} to break page layout.
1193 *
1194 * @param {Array<string>} tags Array of open/close tags (e.g. '<p>', '</p>')
1195 * that will be modified in place to be either an open tag, one or more close
1196 * tags concatenated, or the empty string.
1197 * @return {string} zero or more closed tags that close all elements that are
1198 * opened in tags but not closed.
1199 * @private
1200 */
1201soy.$$balanceTags_ = function(tags) {
1202 var open = [];
1203 for (var i = 0, n = tags.length; i < n; ++i) {
1204 var tag = tags[i];
1205 if (tag.charAt(1) == '/') {
1206 var openTagIndex = goog.array.lastIndexOf(open, tag);
1207 if (openTagIndex < 0) {
1208 tags[i] = ''; // Drop close tag with no corresponding open tag.
1209 } else {
1210 tags[i] = open.slice(openTagIndex).reverse().join('');
1211 open.length = openTagIndex;
1212 }
1213 } else if (tag == '<li>' &&
1214 goog.array.lastIndexOf(open, '</ol>') < 0 &&
1215 goog.array.lastIndexOf(open, '</ul>') < 0) {
1216 // Drop <li> if it isn't nested in a parent <ol> or <ul>.
1217 tags[i] = '';
1218 } else if (!soy.$$HTML5_VOID_ELEMENTS_.test(tag)) {
1219 open.push('</' + tag.substring(1));
1220 }
1221 }
1222 return open.reverse().join('');

1610 * optionally adding ellipsis at the end.

1617 * truncation.
1618 * @return {string} The string after truncation.
1619 */
1620soy.$$truncate = function(str, maxLen, doAddEllipsis) {
1621
1622 str = String(str);
1623 if (str.length <= maxLen) {
1624 return str; // no need to truncate
1625 }
1626
1627 // If doAddEllipsis, either reduce maxLen to compensate, or else if maxLen is
1628 // too small, just turn off doAddEllipsis.
1629 if (doAddEllipsis) {
1630 if (maxLen > 3) {

1632 } else {

1644 str = str.substring(0, maxLen);
1645
1646 // Add ellipsis.
1647 if (doAddEllipsis) {
1648 str += '...';
1649 }
1650
1651 return str;
